    const results = document.getElementById('results');
    const q = document.getElementById('q');
    const go = document.getElementById('go');
    const openAll = document.getElementById('openAll');
    const copyAll = document.getElementById('copyAll');

    function encode(qs){
      return encodeURIComponent(qs.trim()).replace(/%20/g, "+");
    }

    function buildGrid(query){
      results.innerHTML = '';
      const term = query.trim();
      if(!term) { results.innerHTML = '<div class="small">Type a product name to get marketplace links.</div>'; return; }
      const enc = encode(term);
      M.forEach(([name, pattern])=>{
        const url = pattern.replace('%s', enc);
        const card = document.createElement('div');
        card.className = 'card';
        const left = document.createElement('div');
        left.innerHTML = '<div style="font-weight:600">'+name+'</div><div class="small">'+url.replace(/(https?:\/\/[^/]+).*/,'$1')+'</div>';
        const btn = document.createElement('button');
        btn.textContent = 'Open';
        btn.onclick = ()=> window.open(url, '_blank');
        card.appendChild(left);
        card.appendChild(btn);
        results.appendChild(card);
      });
    }

    go.addEventListener('click', ()=> buildGrid(q.value));
    q.addEventListener('keydown', (e)=> { if(e.key==='Enter') buildGrid(q.value); });

    openAll.addEventListener('click', ()=>{
      const term = q.value.trim();
      if(!term){ alert('Type a product first'); return; }
      const enc = encode(term);
      let count = 0;
      M.forEach(([name, pattern])=>{
        const url = pattern.replace('%s', enc);
        window.open(url, '_blank');
        count++;
      });
      if(count===0) alert('No marketplaces configured.');
    });

    copyAll.addEventListener('click', ()=>{
      const term = q.value.trim();
      if(!term){ alert('Type a product first'); return; }
      const enc = encode(term);
      const lines = M.map(([name,pattern]) => pattern.replace('%s', enc));
      navigator.clipboard.writeText(lines.join('\n')).then(()=> alert('All links copied to clipboard.'));
    });

    // initial hint
    buildGrid('');
